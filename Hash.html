    Sort IP addresses numerically by net part, and host names just as usual.

    # File lib/log2counter/core_ext/sort_by_ip_or_host.rb, line 37
def sort_by_ip_or_host
  sort_by { |key, _|
    key ? key.split('.').map { |part|
      part =~ /\A\d+\z/ ? part.to_i : part
    } : []
  }
end
